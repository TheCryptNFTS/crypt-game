===== src/types/faction.ts =====
export type Faction =
  | "STONE"
  | "IRON"
  | "BRONZE"
  | "SILVER"
  | "GOLD"
  | "GOD";

export function normalizeFaction(value: string | null | undefined): Faction {
  const raw = String(value ?? "").trim();

  const normalized = raw
    .toUpperCase()
    .replace(/[\s-]+/g, "_");

  switch (normalized) {
    case "STONE":
    case "STONE_KEEPER":
    case "STONE_KEEPERS":
      return "STONE";

    case "IRON":
    case "IRON_DEFENDER":
    case "IRON_DEFENDERS":
      return "IRON";

    case "BRONZE":
    case "BRONZE_GUARDIAN":
    case "BRONZE_GUARDIANS":
      return "BRONZE";

    case "SILVER":
    case "SILVER_SENTINEL":
    case "SILVER_SENTINELS":
      return "SILVER";

    case "GOLD":
    case "GOLDEN":
    case "GOLDEN_SOVEREIGN":
    case "GOLDEN_SOVEREIGNS":
      return "GOLD";

    case "GOD":
    case "GODS":
      return "GOD";

    default:
      throw new Error(`Unknown faction: ${value}`);
  }
}
